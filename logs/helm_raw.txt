**opentelemetry-helm-chart.yaml**
```yaml
# Source: opentelemetry/templates/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: opentelemetry
spec:
  replicas: 1
  selector:
    matchLabels:
      app: opentelemetry
  template:
    metadata:
      labels:
        app: opentelemetry
    spec:
      containers:
      - name: opentelemetry
        image: otel/opentelemetry-collector:latest
        ports:
        - containerPort: 55678
---
# Source: opentelemetry/templates/service.yaml
apiVersion: v1
kind: Service
metadata:
  name: opentelemetry
spec:
  selector:
    app: opentelemetry
  ports:
  - name: http
    port: 55678
    targetPort: 55678
  type: ClusterIP
---
# Source: opentelemetry/templates/configmap.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: opentelemetry-config
data:
  config.yaml: |
    receivers:
    - protocol: http
      host: 0.0.0.0
      port: 55678
    exporters:
    - logging:
        loglevel: debug
    service:
      pipelines:
      - receivers:
        - http
        exporters:
        - logging
---
# Source: opentelemetry/templates/values.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: opentelemetry-values
data:
  config.yaml: |
    global:
      telemetry:
        endpoint: http://opentelemetry:55678
```
